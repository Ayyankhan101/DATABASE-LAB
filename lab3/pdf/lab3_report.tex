\subsection{Query 1: Overall Business Summary}

This query computes key performance indicators (KPIs) for delivered orders.

\begin{lstlisting}[caption={Overall Business Summary Query}, label=lst:q1]
SELECT COUNT(*) AS total_orders,
       SUM(total_amount) AS total_revenue,
       ROUND(AVG(total_amount), 2) AS avg_order_value,
       MAX(total_amount) AS largest_order
FROM orders
WHERE status = 'delivered';
\end{lstlisting}

\noindent\textbf{Techniques:} Aggregate functions, \texttt{WHERE} clause, \texttt{ROUND}

\subsection{Query 2: Revenue by Product Category}

Analyzes revenue contribution across product categories using joins and aggregation.

\begin{lstlisting}[caption={Revenue by Product Category Query}, label=lst:q2]
SELECT p.category,
       COUNT(DISTINCT oi.order_id) AS orders_containing,
       SUM(oi.quantity * oi.unit_price) AS category_revenue
FROM order_items oi
JOIN products p ON oi.product_id = p.product_id
GROUP BY p.category
ORDER BY category_revenue DESC;
\end{lstlisting}

\noindent\textbf{Techniques:} \texttt{JOIN}, \texttt{GROUP BY}, \texttt{DISTINCT}, aggregation

\subsection{Query 4: Active Cities (HAVING Clause)}

Identifies cities with more than 2 orders using the \texttt{HAVING} clause.

\begin{lstlisting}[caption={Active Cities Query}, label=lst:q4]
SELECT c.city,
       COUNT(DISTINCT c.customer_id) AS customers,
       COUNT(o.order_id) AS total_orders
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.city
HAVING COUNT(o.order_id) > 2
ORDER BY total_orders DESC;
\end{lstlisting}

\noindent\textbf{Techniques:} \texttt{HAVING} vs \texttt{WHERE}, multi-column aggregation

\subsection{Query 5: Device Performance with HAVING}

Analyzes user session metrics by device type with conditional aggregation.

\begin{lstlisting}[caption={Device Performance Query}, label=lst:q5]
SELECT device,
       COUNT(*) AS total_sessions,
       ROUND(AVG(duration_mins), 2) AS avg_duration,
       SUM(CASE WHEN converted THEN 1 ELSE 0 END) AS conversions,
       ROUND(100.0 * SUM(CASE WHEN converted THEN 1 ELSE 0 END) / COUNT(*), 1) 
           AS conversion_rate_pct
FROM user_sessions
GROUP BY device
HAVING COUNT(*) >= 5
ORDER BY avg_duration DESC;
\end{lstlisting}

\noindent\textbf{Techniques:} Conditional aggregation (\texttt{CASE}), percentage calculations

\subsection{Query 6: Ranking Orders with ROW\_NUMBER}

Demonstrates window functions to rank orders within each customer.

\begin{lstlisting}[caption={Ranking Orders with ROW_NUMBER}, label=lst:q6]
SELECT customer_id,
       order_id,
       order_date,
       total_amount,
       ROW_NUMBER() OVER (
           PARTITION BY customer_id
           ORDER BY total_amount DESC
       ) AS rank_within_customer
FROM orders
ORDER BY customer_id, rank_within_customer;
\end{lstlisting}

\noindent\textbf{Techniques:} Window functions, \texttt{ROW\_NUMBER()}, \texttt{PARTITION BY}

\subsubsection*{Query 6b: Each Customer's Highest-Value Order}

Using CTE with \texttt{ROW\_NUMBER} to find top orders.

\begin{lstlisting}[caption={Highest-Value Order per Customer}, label=lst:q6b]
WITH ranked_orders AS (
    SELECT customer_id,
           order_id,
           order_date,
           total_amount,
           ROW_NUMBER() OVER (
               PARTITION BY customer_id
               ORDER BY total_amount DESC
           ) AS rank_within_customer
    FROM orders
)
SELECT customer_id, order_id, order_date, total_amount
FROM ranked_orders
WHERE rank_within_customer = 1
ORDER BY customer_id;
\end{lstlisting}

\subsection{Query 7: Overall Revenue Ranking with RANK}

Compares \texttt{RANK} vs \texttt{DENSE\_RANK} for order revenue ranking.

\begin{lstlisting}[caption={Revenue Ranking with RANK and DENSE_RANK}, label=lst:q7]
SELECT order_id,
       customer_id,

       RANK() OVER (ORDER BY total_amount DESC) AS rank,
       DENSE_RANK() OVER (ORDER BY total_amount DESC) AS dense_rank
FROM orders

\subsection{Query 8: Month-over-Month Revenue Trend with LAG}

Uses the \texttt{LAG} window function to compute period-over-period changes.

\begin{lstlisting}[caption={Month-over-Month Trend with LAG}, label=lst:q8]
WITH monthly_revenue AS (
    SELECT TO_CHAR(order_date, 'YYYY-MM') AS month,
           SUM(total_amount) AS revenue
    FROM orders
    WHERE status = 'delivered'
    GROUP BY TO_CHAR(order_date, 'YYYY-MM')
)
SELECT month,
       revenue,
       LAG(revenue) OVER (ORDER BY month) AS prev_month,
       revenue - LAG(revenue) OVER (ORDER BY month) AS absolute_change,
       ROUND(
           100.0 * (revenue - LAG(revenue) OVER (ORDER BY month))
           / NULLIF(LAG(revenue) OVER (ORDER BY month), 0),
           1
       ) AS pct_change
FROM monthly_revenue
ORDER BY month;
\end{lstlisting}

\noindent\textbf{Techniques:} CTEs, \texttt{LAG()}, null-safe division (\texttt{NULLIF})

\subsection{Query 9: Customer Value Segmentation using CTEs}

Multi-level CTE for customer tier classification and revenue analysis.

\begin{lstlisting}[caption={Customer Value Segmentation}, label=lst:q9]
WITH customer_spend AS (
    SELECT c.customer_id,
           c.name,
           c.city,
           COALESCE(SUM(o.total_amount), 0) AS total_spent
    FROM customers c
    LEFT JOIN orders o ON c.customer_id = o.customer_id
    AND o.status = 'delivered'
    GROUP BY c.customer_id, c.name, c.city
),
customer_tiers AS (
    SELECT *,
           CASE
               WHEN total_spent > 30000 THEN 'VIP'
               WHEN total_spent > 10000 THEN 'High Value'
               WHEN total_spent > 0 THEN 'Active'
               ELSE 'Never Purchased'
           END AS tier
    FROM customer_spend
)
SELECT tier,
       COUNT(*) AS customers,
       SUM(total_spent) AS total_revenue,
       ROUND(100.0 * SUM(total_spent) / SUM(SUM(total_spent)) OVER (), 1) 
           AS pct_of_revenue
FROM customer_tiers
GROUP BY tier
ORDER BY total_revenue DESC;
\end{lstlisting}

\noindent\textbf{Techniques:} Multi-level CTEs, \texttt{COALESCE}, window aggregates

\subsection{Query 10: Session-to-Purchase Funnel Analysis}

Complex CTE combining session and order data for funnel metrics.

\begin{lstlisting}[caption={Session-to-Purchase Funnel Analysis}, label=lst:q10]
WITH session_summary AS (
    SELECT customer_id,
           COUNT(*) AS total_sessions,
           SUM(pages_viewed) AS total_pages,
           ROUND(AVG(duration_mins), 2) AS avg_duration,
           SUM(CASE WHEN converted THEN 1 ELSE 0 END) AS converted_sessions
    FROM user_sessions
    GROUP BY customer_id
),
order_summary AS (
    SELECT customer_id,
           COUNT(*) AS total_orders,
           SUM(total_amount) AS total_spent
    FROM orders
    WHERE status = 'delivered'
    GROUP BY customer_id
),
combined AS (
    SELECT c.name,
           c.city,
           s.total_sessions,
           s.total_pages,
           s.avg_duration,
           s.converted_sessions,
           COALESCE(o.total_orders, 0) AS total_orders,
           COALESCE(o.total_spent, 0) AS total_spent
    FROM session_summary s
    JOIN customers c ON s.customer_id = c.customer_id
    LEFT JOIN order_summary o ON s.customer_id = o.customer_id
)
SELECT name,
       city,
       total_sessions,
       total_pages,
       avg_duration,
       total_orders,
       total_spent,
       ROUND(
           CASE WHEN total_sessions > 0
                THEN 100.0 * total_orders / total_sessions
                ELSE 0
           END, 1
       ) AS orders_per_100_sessions
FROM combined
ORDER BY total_sessions DESC, orders_per_100_sessions ASC;
\end{lstlisting}

\noindent\textbf{Techniques:} Multiple CTEs, funnel analysis, conversion metrics
